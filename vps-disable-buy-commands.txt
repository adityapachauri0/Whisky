echo "Finding website directory..."
find /var/www -name "*.html" -type f | head -5
find /var/www -type d -name "*viticult*"
ls -la /var/www/

# Step 2: Navigate to the most likely website directory
cd /var/www/viticultwhisky.co.uk || cd /var/www/html || cd /var/www/whisky/frontend/build
